# Comandos útiles para Docker

# Construir la imagen
docker build -t soflex-desktop .

# Levantar el contenedor (puedes cambiar el nombre y el puerto si levantas varios)
docker run -d --name soflex1 -p 5900:5900 soflex-desktop

# Ver logs del contenedor
docker logs soflex1

# Acceder a una terminal dentro del contenedor
docker exec -it soflex1 bash

# Parar el contenedor
docker stop soflex1

# Eliminar el contenedor
docker rm soflex1

# Eliminar la imagen
docker rmi soflex-desktop

# Listar contenedores activos
docker ps

# Listar todos los contenedores (incluidos los detenidos)
docker ps -a

# Listar imágenes
docker images

# Usar Docker sin sudo (agrega tu usuario al grupo docker)
sudo usermod -aG docker $USER
# Luego cierra sesión y vuelve a entrar, o ejecuta:
newgrp docker
